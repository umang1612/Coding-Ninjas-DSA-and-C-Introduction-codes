*  output - Save the result in the output array (passed as argument). You donâ€™t have to 
*  print or return the result
*/
void minLengthWord(char input[], char output[]){
    int w=0,count=0;
    while(input[w]!=0){
        count+=1;
        w+=1;
    }
    int start=0,end,length=10000000000,wordlength,outputstart=0,outputend=0,p=0;
    for(int i=0;input[i]!='\0';i++){
        if(input[i]==' '||i==count-1){
            if(i==count-1){
                end=i;
            }
            else{
                end=i-1;}
            wordlength=end-start;
            if(wordlength<length){
                outputstart=start;
                outputend=end;
                length=wordlength;
            }
            
            start=i+1;
    }
    }
		
	while(outputstart<=outputend){
        output[p]=input[outputstart];
        p+=1;
        outputstart+=1;
    }	

}
